#include<bits/stdc++.h>

using namespace std;
int dp[7000][7000];
string s,s1;
int n;
int find_sub(int i,int j)
{
     if(i>=j)return 1;
     if(dp[i][j]!=-1) return dp[i][j];
     if(s[i]==s[j])
     {
         return dp[i][j]=find_sub(i+1,j-1);
     }
     else
     {
         return dp[i][j]=0;
     }
}


int main()
{
    int t;cin>>t;
    while(t--)
    {
        cin>>s;
        n=s.size();
        int mx=INT_MIN;
        memset(dp,-1,sizeof dp);
        for(int i=0;i<n;i++)
        {
          for(int j=i;j<n;j++)
           {
               if(find_sub(i,j))
               mx=max(mx,j-i+1)
            }
        }
        cout<<mx<<endl;
        
    }
}
